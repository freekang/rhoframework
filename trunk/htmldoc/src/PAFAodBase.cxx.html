PAFAodBase::PAFAodBase() :
_xposition(0),
_yposition(0),
_zposition(0),
_xmomentum(0),
_ymomentum(0),
_zmomentum(0),
_energy(0),
_mcCons(0),
_mcHypo(0),
_mcIndex(-1),
_DetBitpattern(0),
_charge(-1)
{
    for (int j=0; j<MATRIXSIZE;j++) {
	_errP7[j] = 0;
    }
    for (Int_t i=0; i<NELEM_AODBASE; i++) {
	fBranches[i]=0;
    }
    
}

PAFAodBase::PAFAodBase(PAFAodBase& other)
{
    // set branch pointers of copy 
    fBranches[0]=NULL;
    Int_t i;
    for (i=1; i<NELEM_AODBASE; i++) {
	if (other.fBranches[i]!=other.fBranches[0]) {
	    fBranches[i]=other.fBranches[i];
	} else {
	    fBranches[i] = fBranches[0];
	}
    }
    // copy data
    _xposition = other._xposition;
    _yposition = other._yposition;
    _zposition = other._zposition;
    _xmomentum = other._xmomentum;
    _ymomentum = other._ymomentum;
    _zmomentum = other._zmomentum;
    _energy = other._energy;
    _mcCons = other._mcCons;
    _mcHypo = other._mcHypo;
    _mcIndex = other._mcIndex;
    _DetBitpattern = other._DetBitpattern;
    _charge = other._charge;
    for (i=0; i<MATRIXSIZE; i++) {
	_errP7[i] = other._errP7[i];
    }
}

 PAFAodBase::~PAFAodBase()  
{ 
    for (int i=0; i<NELEM_AODBASE; i++) delete fBranches[i];
}

 void PAFAodBase::PrintOn(std::ostream &o) const
{
    TVector3 r = GetPosition();
    TVector3 p = GetMomentum();
    o << " FIT position                     : (" << r.X() << ";" << r.Y() << ";" << r.Z() << ")" << endl;
    o << " FIT momentum                     : (" << p.X() << ";" << p.Y() << ";" << p.Z() << ")" << endl;
    o << " FIT momentum.mag                 : " << p.Mag() << endl;
    o << " FIT charge                       : " << GetCharge() << endl;
    o << " FIT energy                       : " << GetEnergy() << endl;
    const Float_t *err = GetPositionError();
    o << " FIT postion error                :   posErr(1,1) = " << err[0] << " ; posErr(2,2) = " 
	<< err[2] << " ; posErr(3,3) = " << err[5] << endl;
    o << " FIT mtm error                    :   mtmErr(1,1) = " << err[18] << " ; mtmErr(2,2) = " 
	<< err[20] << " ; mtmErr(3,3) = " << err[23] << " ; mtmErr(4,4) = " << err[27] << endl;
}


 void PAFAodBase::Init(PAFAodBase* other)
{
    
    for (Int_t i=1; i<NELEM_AODBASE; i++) {
	fBranches[i] = other->fBranches[i];
    }
}

void
 PAFAodBase::Init(TBranchClones *clonesBranch)
{
    TString branchName(clonesBranch->GetName());
    TBranch *subBranch;
    TObjArray *list = clonesBranch->GetListOfBranches();
    Int_t listLen = list->GetEntries();
    for (Int_t i=0; i<listLen; i++) {
	subBranch = (TBranch*) (*list)[i];
	TString t(subBranch->GetName());
	
	if      (t == branchName+"._xposition") { fBranches[1] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._yposition") { fBranches[2] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._zposition") { fBranches[3] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._xmomentum") { fBranches[4] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._ymomentum") { fBranches[5] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._zmomentum") { fBranches[6] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._energy") { fBranches[7] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._errP7[28]") { fBranches[8] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._mcCons") { fBranches[9] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._mcHypo") { fBranches[10] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._mcIndex") { fBranches[11] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DetBitpattern") { fBranches[12] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._charge") { fBranches[13] = new TSmartPointer(subBranch); }
    }
}
