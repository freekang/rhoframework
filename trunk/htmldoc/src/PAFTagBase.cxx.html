 TLorentzVector PAFTagBase::GetCMFrame()  const {
    if (GetFloat(fBranches[4],_cmf4)>10.)
    return TLorentzVector(GetFloat(fBranches[1],_cmf1),
			  GetFloat(fBranches[2],_cmf2),
			  GetFloat(fBranches[3],_cmf3),
			  GetFloat(fBranches[4],_cmf4) );
    else
    return TLorentzVector(-0.12,0.,5.87,12.1);
}

    TString theKey = prefix + key;
