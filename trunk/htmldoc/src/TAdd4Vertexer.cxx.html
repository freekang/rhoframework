TLorentzVector 
 TAdd4Vertexer::P4() const 
{ 
    return fP4; 
}

TMatrixD&
 TAdd4Vertexer::P4Err() 
{ 
    return fErr; 
}

VAbsVertex* 
 TAdd4Vertexer::Compute( TCandListIterator& iter )
{
    fP4  = TLorentzVector();
    fErr.Zero();
    TCandidate* cand=0;
    while( cand=iter.Next() )
    {
	fP4 +=cand->P4();
	fErr+=cand->P4Err();
    }
    return TFactory::Instance()->NewVertex();
}
