 void BtaCandidateR::PrintOn(ostream &o) const
{
    //Vertex
    if (DecayVtx() != 0) {
       VAbsVertex *vtx = DecayVtx()->Transient();
       o << "Vertex " << DecayVtx()->refToMe().id() << ": " << *vtx;
    }	

    // Id
    UInt_t myId = refToMe().id();
    o << "Candidate "<< myId << ": ";

    // Daughters
    for (int n=0;n<NDaughters();++n)
    {
	o << " "<< Daughter(n);
    }

    o << endl;
}
