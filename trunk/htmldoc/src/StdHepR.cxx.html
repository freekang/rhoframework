 void StdHepR::initialiseArray() {

  // Initialise TClonesArray:
  if( _theArray == 0 ) {
    _array= new TClonesArray( "StdHepTrkR", 100 );
  }
  else {
    _array= _theArray;
    _ownArray= kFALSE;
  }

  _array->BypassStreamer(kFALSE);

}


//--------------
// Destructor --
//--------------
 StdHepR::~StdHepR() {

  _array->Clear();
  if( _ownArray ) delete _array;

}
