 PAFAodTrack::PAFAodTrack(PAFAodTrack &other)
{
    // set branch pointers of copy 
    fBranches[0]=NULL;
    Int_t i;
    for (i=1; i<NELEM_AODTRACK; i++) {
	if (other.fBranches[i]!=other.fBranches[0]) {
	    fBranches[i]=other.fBranches[i];
	} else {
	    fBranches[i] = fBranches[0];
	}
    }
    // copy data
    for (i=0; i<5; i++) {
      _DRCnexp[i] = other._DRCnexp[i];
    }
    _SVThitpattern = other._SVThitpattern;
    _SVThits = other._SVThits;
    _DCHinnerlayer = other._DCHinnerlayer;
    _DCHouterlayer = other._DCHouterlayer;
    _DCHhits = other._DCHhits;
    _TRKpidhypo = other._TRKpidhypo;
    _TRKndof = other._TRKndof;
    _TRKfitstatus = other._TRKfitstatus;
    _TRKprob = other._TRKprob;
    _TRKchi2 = other._TRKchi2;
    _TRKtlen = other._TRKtlen;
    _DRCthetaC = other._DRCthetaC;
    _DRCethetaC = other._DRCethetaC;
    _DCHmom = other._DCHmom;
    _DRCmom = other._DRCmom;
    _DRCnphot = other._DRCnphot;
    _DRCnbkgd = other._DRCnbkgd;
    _DRCstatus = other._DRCstatus; 
    _DRCnumbar = other._DRCnumbar;
    _DRCexit = other._DRCexit;
    _DRCxpos = other._DRCxpos;
    _DCHdedx = other._DCHdedx;
    _SVTdedx = other._SVTdedx;
    _SVTPidInfo = other._SVTPidInfo;
    _DCHPidInfo = other._DCHPidInfo;
    _DRCPidInfo = other._DRCPidInfo;
    _pattern = other._pattern; 
    _TRKst = other._TRKst;
    _DCHsamples = other._DCHsamples;
    _SVTsamples = other._SVTsamples;
    _EMCintertheta = other._EMCintertheta;
    _EMCinterphi = other._EMCinterphi;
}

 PAFAodTrack::~PAFAodTrack()  
{ 
    for (int i=0; i<NELEM_AODTRACK; i++) delete fBranches[i];
}

//Detector status ( = subsystem with hits) __________

 Bool_t PAFAodTrack::SvtOk() const {
    if (fBranches[36]) fBranches[36]->CheckBranch();
    return ((_pattern&1)==1);
}

 Bool_t PAFAodTrack::DchOk() const {
    if (fBranches[36]) fBranches[36]->CheckBranch();
    return ((_pattern&2)==2);
}

 Bool_t PAFAodTrack::DrcOk() const {
    if (fBranches[36]) fBranches[36]->CheckBranch();
    return ((_pattern&4)==4);
}


 void PAFAodTrack::SetTrackStatus(Bool_t svt,Bool_t dch, Bool_t drc)
{
    _pattern=0;
    if (svt) { _pattern |=1; }
    if (dch) { _pattern |=2; }
    if (drc) { _pattern |=4; }
}

void
 PAFAodTrack::PrintOn(std::ostream &o) const {
    if (_pattern&1) {
	o << " SVT #hits                        : " << (Int_t) GetSvtHits() << endl;
	o << " SVT pattern                      : " << (Int_t) GetSvtHitPattern() << endl;
	o << " SVT dE/dx                        : " << GetSvtMeanDEdx() << endl;
    } else {
	o << " SVT : no track " << endl;
    }
    
    if (_pattern&2) {
	o << " DCH #hits                        : " << (Int_t) GetDchHits() << endl;
	o << " DCH inner/outer layer            : " << (Int_t) GetDchInnerLayer() << " / " << (Int_t) GetDchOuterLayer()  << endl;
	o << " DCH dE/dx                        : " << GetDchMeanDEdx() << endl; 
	o << " DCH momentum correction          : " << GetDchDeltaMomentum() << endl; 
    } else {
	o << " DCH : no track  " << endl;
    }
    
    o << " TRK pid hypo                     : " << (Int_t) GetChargedPidHypo() << endl;
    o << " TRK ndof                         : " << GetDegreesOfFreedom() << endl;
    o << " TRK chi**2                       : " << GetChiSquared() << endl;
    o << " TRK length                       : " << GetTrackLength() << endl;
    o << " TRK start found range            : " << GetTrackStartFoundRange() << endl;
    
    if (_pattern&4) {
	o << " DRC thetaC                     : " << GetDrcThetaC() << endl;
	o << " DRC thetaC error               : " << GetDrcThetaCError() << endl;
	o << " DRC number of photons          : " << GetDrcNumberOfPhotons() << endl;
	o << " DRC number of bkgd photons     : " << GetDrcNumberOfBackgroundPhotons() << endl;
	o << " DRC expected Photons (e)       : " << GetDrcNumberOfExpectedPhotons(0) << endl;
	o << " DRC expected Photons (mu)      : " << GetDrcNumberOfExpectedPhotons(1) << endl;
	o << " DRC expected Photons (pi)      : " << GetDrcNumberOfExpectedPhotons(2) << endl;
	o << " DRC expected Photons (k)       : " << GetDrcNumberOfExpectedPhotons(3) << endl;
	o << " DRC expected Photons (p)       : " << GetDrcNumberOfExpectedPhotons(4) << endl;
	o << " DRC Entance Bar                : " << GetDrcNumBar() << endl;
	o << " DRC Exit    Bar                : " << GetDrcExit() << endl;
	o << " DRC xpos                       : " << GetDrcXpos() << endl;
	o << " DRC delta Momentum             : " << GetDrcDeltaMomentum() << endl;
    } else {
	o << " DRC : no track " << endl;
    }
    o << " EMC - TRK intersection : theta = " << GetDrcThetaAtEmc() << " phi = " << _EMCinterphi << endl;
    GetDchPidInfo().PrintOn(cout);
    GetSvtPidInfo().PrintOn(cout);
    GetDrcPidInfo().PrintOn(cout);
    
    
}

 void PAFAodTrack::Init(PAFAodTrack* other)
{
    for (Int_t i=1; i<NELEM_AODTRACK; i++) {
	fBranches[i] = other->fBranches[i];
    }
}

void
 PAFAodTrack::Init(TBranchClones *clonesBranch)
{
    TString branchName(clonesBranch->GetName());
    TBranch *subBranch;
    TObjArray *list = clonesBranch->GetListOfBranches();
    Int_t listLen = list->GetEntries();
    for (Int_t i=0; i<listLen; i++) {
	subBranch = (TBranch*) (*list)[i];
	TString t(subBranch->GetName());
	// SVT
	if      (t == branchName+"._SVThitpattern") { fBranches[1] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._SVThits") { fBranches[2] = new TSmartPointer(subBranch); }
	// DCH
	else if (t == branchName+"._DCHinnerlayer") { fBranches[3] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHouterlayer") { fBranches[4] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHhits") { fBranches[5] = new TSmartPointer(subBranch); }
	// TRK
	else if (t == branchName+"._TRKpidhypo") { fBranches[6] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._TRKndof") { fBranches[7] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._TRKfitstatus") { fBranches[8] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._TRKprob") { fBranches[9] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._TRKchi2") { fBranches[10] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._TRKtlen") { fBranches[11] = new TSmartPointer(subBranch); }
	// PIDqual
	else if (t == branchName+"._DRCthetaC") { fBranches[12] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCethetaC") { fBranches[13] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHmom") { fBranches[14] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCmom") { fBranches[15] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCnphot") { fBranches[16] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCnbkgd") { fBranches[17] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCstatus") { fBranches[18] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCnumbar") { fBranches[19] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCexit") { fBranches[20] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCxpos") { fBranches[21] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHdedx") { fBranches[22] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._SVTdedx") { fBranches[23] = new TSmartPointer(subBranch); }
	// DchPidInfo
	else if (t == branchName+"._DCHPidInfo._likelihood[5]") { fBranches[24] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHPidInfo._significanceLevel[5]") { fBranches[25] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHPidInfo._status") { fBranches[26] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHPidInfo._sign") { fBranches[27] = new TSmartPointer(subBranch); }
	// DrcPidInfo
	else if (t == branchName+"._DRCPidInfo._likelihood[5]") { fBranches[28] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCPidInfo._significanceLevel[5]") { fBranches[29] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCidInfo._status") { fBranches[30] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCPidInfo._sign") { fBranches[31] = new TSmartPointer(subBranch); }
	// SvtPidInfo
	else if (t == branchName+"._SVTPidInfo._likelihood[5]") { fBranches[32] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._SVTPidInfo._significanceLevel[5]") { fBranches[33] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._SVTidInfo._status") { fBranches[34] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._SVTPidInfo._sign") { fBranches[35] = new TSmartPointer(subBranch); }

	else if (t == branchName+"._pattern") { fBranches[36] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._TRKst") { fBranches[37] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DRCnexp[5]") { fBranches[38] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._DCHsamples") { fBranches[39] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._SVTsamples") { fBranches[40] = new TSmartPointer(subBranch); }

	else if (t == branchName+"._EMCinterphi") { fBranches[41] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCintertheta") { fBranches[42] = new TSmartPointer(subBranch); }

	// Unknown branch
	//else 
	//    cout << "PAFAodTrack::Init: Unknown branch " << t << endl;
    }
}
