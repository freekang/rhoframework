 PAFAodCluster::PAFAodCluster(PAFAodCluster &other)
{
    // set branch pointers of copy 
    fBranches[0]=NULL;
    Int_t i;
    for (i=1; i<NELEM_AODCLUSTER; i++) {
	if (other.fBranches[i]!=other.fBranches[0]) {
	    fBranches[i]=other.fBranches[i];
	} else {
	    fBranches[i] = fBranches[0];
	}
    }
    // copy Data
    _EMCcrystals = other._EMCcrystals;
    _EMCbumps = other._EMCbumps;
    _EMCenergy = other._EMCenergy;
    _EMClateralshape = other._EMClateralshape;
    _EMC20 = other._EMC20;
    _EMC31 = other._EMC31;
    _EMC42 = other._EMC42;
    _EMCsecmom = other._EMCsecmom;
    _EMCconsistency = other._EMCconsistency;
    _EMCcalenergy = other._EMCcalenergy;
    _EMCstatus = other._EMCstatus;
    _EMCpidhypo = other._EMCpidhypo;
    _EMCcx = other._EMCcx;
    _EMCcy = other._EMCcy;
    _EMCcz = other._EMCcz;
    _EMCcovenergy = other._EMCcovenergy;
    _EMCcovtheta = other._EMCcovtheta;
    _EMCcovphi = other._EMCcovphi;
    _EMCcovrho = other._EMCcovrho;
    _IFRlayers = other._IFRlayers;
    _IFRstrips = other._IFRstrips;
    _IFRpattern = other._IFRpattern;
    _IFRclusters = other._IFRclusters;
    _IFRfirsthit = other._IFRfirsthit;
    _IFRlasthit = other._IFRlasthit;
    _IFRiron = other._IFRiron;
    _IFRtrk = other._IFRtrk;
    _IFRemc = other._IFRemc;
    _IFRtrkmatch = other._IFRtrkmatch;
    _IFRemcmatch = other._IFRemcmatch;
    for (i=0; i<20; i++) {
	_IFRlaystrips[i] = other._IFRlaystrips[i];
    }
    _IFRrpcMuProb = other._IFRrpcMuProb;
    _IFRclusfitChi2 = other._IFRclusfitChi2;
    _cpattern = other._cpattern;
    _EMCPidInfo = other._EMCPidInfo;
    _IFRPidInfo = other._IFRPidInfo;
    _EMCs9s25 = other._EMCs9s25;
    _EMCs1s9 = other._EMCs1s9;
    _EMClikelihood = other._EMClikelihood;
    _EMCbits = other._EMCbits;
    _IFRlastbar = other._IFRlastbar;
}

 PAFAodCluster::~PAFAodCluster()  
{ 
    for (int i=0; i<NELEM_AODCLUSTER; i++) delete fBranches[i];
}

//Detector status ( = subsystem with hits) __________


 Bool_t PAFAodCluster::EmcOk() const {
    if (fBranches[34]) fBranches[34]->CheckBranch();
    return ((_cpattern&8)==8);
}

 Bool_t PAFAodCluster::IfrOk() const {
    if (fBranches[34]) fBranches[34]->CheckBranch();
    return ((_cpattern&16)==16);
}

 void PAFAodCluster::SetClusterStatus(Bool_t emc, Bool_t ifr) {
    _cpattern=0;
    if (emc) { _cpattern |=8; }
    if (ifr) { _cpattern |=16; }
}

 void PAFAodCluster::PrintOn(std::ostream &o) const 
{
    if (_cpattern&8) {
	o << " EMC raw energy :                     : " << GetEmcRawEnergy() << endl;
	o << " EMC cal energy :                     : " << GetEmcCalEnergy() << endl;
	o << " Number of Crystals in EmcCluster : " << GetEmcNumberOfCrystals() << endl;
	o << " EMC lateral shape                : " << GetEmcLateralShape() << endl;
	o << " EMC Zernike moment (2,0)         : " << GetEmcZernikeMoment20() << endl;
	o << " EMC Zernike moment (3,1)         : " << GetEmcZernikeMoment31() << endl;
	o << " EMC Zernike moment (4,2)         : " << GetEmcZernikeMoment42() << endl;
	o << " EMC second moment                : " << GetEmcSecondMoment() << endl;
	TVector3 r = GetEmcCentroid();
	o << " EMC cluster position :           : (" << r.X() << ";" << r.Y() << ";" << r.Z() <<")" << endl;
    } else {
	o << " EMC : no Cluster " << endl;
    }
    
    if (_cpattern&16) {
	o << " IFR #layers hit                  : " << (Int_t) GetIfrHitLayers() << endl;
	o << " IFR #strips                      : " << (Int_t) GetIfrNumberOfStrips() << endl;
	o << " IFR clusters                     : " << (Int_t) GetIfrClusters() << endl;
	o << " IFR first/last hit layer         : " << (Int_t) GetIfrFirstHit() << " / " << (Int_t) _IFRlasthit << endl;
	o << " IFR expected                     : " << GetIfrExpectedInteractionLengths() << endl;
	o << " IFR measured                     : " << GetIfrMeasuredInteractionLengths() << endl;
	o << " IFR before iron                  : " << GetIfrInteractionLengthsBeforeIron() << endl;
	o << " IFR track match                  : " << GetIfrTrackMatchConsistency() << endl;
	o << " IFR EMC match                    : " << GetIfrEmcMatchConsistency() << endl;
	o << " IFR rpc muon probability         : " << GetIfrRpcMuProb() << endl;
	o << " IFR chi**2 of cluster fit        : " << GetIfrClusfitChi2() << endl;
	o << " IFR last barrel                  : " << GetIfrLastBarrel() << endl;
    } else {
	o << " IFR : no Cluster " << endl;
    }
    _EMCPidInfo.PrintOn(cout);
    _IFRPidInfo.PrintOn(cout);

} 

 void PAFAodCluster::Init(PAFAodCluster* other)
{
    for (Int_t i=1; i<NELEM_AODCLUSTER; i++) {
	fBranches[i] = other->fBranches[i];
    }
}

void
 PAFAodCluster::Init(TBranchClones *clonesBranch)
{
    TString branchName(clonesBranch->GetName());
    TBranch *subBranch;
    TObjArray *list = clonesBranch->GetListOfBranches();
    Int_t listLen = list->GetEntries();
    for (Int_t i=0; i<listLen; i++) {
	subBranch = (TBranch*) (*list)[i];
	TString t(subBranch->GetName());
	// EMC
	if      (t == branchName+"._EMCcrystals") { fBranches[1] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCbumps") { fBranches[2] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCenergy") { fBranches[3] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMClateralshape") { fBranches[4] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMC20") { fBranches[5] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMC31") { fBranches[6] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMC42") { fBranches[7] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCsecmom") { fBranches[8] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCconsistency") { fBranches[9] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCcalenergy") { fBranches[10] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCstatus") { fBranches[11] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCpidhypo") { fBranches[12] = new TSmartPointer(subBranch); } 
	else if (t == branchName+"._EMCcx") { fBranches[13] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCcy") { fBranches[14] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCcz") { fBranches[15] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCcovenergy") { fBranches[16] = new TSmartPointer(subBranch); } 
	else if (t == branchName+"._EMCcovtheta") { fBranches[17] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCcovphi") { fBranches[18] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCcovrho") { fBranches[19] = new TSmartPointer(subBranch); }   
	// IFR
	else if (t == branchName+"._IFRlayers") { fBranches[20] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRstrips") { fBranches[21] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRpattern") { fBranches[22] = new TSmartPointer(subBranch); } 
	else if (t == branchName+"._IFRclusters") { fBranches[23] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRfirsthit") { fBranches[24] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRlasthit") { fBranches[25] = new TSmartPointer(subBranch); } 
	else if (t == branchName+"._IFRiron") { fBranches[26] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRtrk") { fBranches[27] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRemc") { fBranches[28] = new TSmartPointer(subBranch); }    
	else if (t == branchName+"._IFRtrkmatch") { fBranches[29] = new TSmartPointer(subBranch); }  
	else if (t == branchName+"._IFRemcmatch") { fBranches[30] = new TSmartPointer(subBranch); }    
	else if (t == branchName+"._IFRlaystrips[20]") { fBranches[31] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._IFRrpcMuProb") { fBranches[32] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._IFRclusfitChi2") { fBranches[33] = new TSmartPointer(subBranch); }
	// pattern
	else if (t == branchName+"._cpattern") { fBranches[34] = new TSmartPointer(subBranch); }        
	// EmcPidInfo
	else if (t == branchName+"._EMCPidInfo._likelihood[5]") { fBranches[35] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCPidInfo._significanceLevel[5]") { fBranches[36] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCidInfo._status") { fBranches[37] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCPidInfo._sign") { fBranches[38] = new TSmartPointer(subBranch); }
	// IfrPidInfo
	else if (t == branchName+"._IFRPidInfo._likelihood[5]") { fBranches[39] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._IFRPidInfo._significanceLevel[5]") { fBranches[40] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._IFRidInfo._status") { fBranches[41] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._IFRPidInfo._sign") { fBranches[42] = new TSmartPointer(subBranch); }
	// new in 8.6.x
	else if (t == branchName+"._EMCs9s25") { fBranches[43] = new TSmartPointer(subBranch); }
	else if (t == branchName+"._EMCs1s9") { fBranches[44] = new TSmartPointer(subBranch); }
