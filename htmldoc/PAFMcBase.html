    const Pdt *pdt = Pdt::Instance();
    PdtEntry* pdtentry = pdt->lookup( static_cast<const PdtLund::LundType>( fPdtID ) ); // RD!! a pitty for OSF, the const is nessecary (a Sun Bug)
